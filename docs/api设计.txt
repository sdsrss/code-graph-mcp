  ---
  Context Sandbox 触发逻辑

  semantic_code_search 内部自动判断：
  - 结果集总 token 数 ≤ 阈值（如 2000 tokens）→ 直接返回完整代码
  - 结果集总 token 数 > 阈值 → 触发 Sandbox 压缩：
    - 返回每个结果的 {name, signature, relevance_score, summary(前3行), pointer_id}
    - Claude 按需调用 read_snippet(pointer_id) 获取完整代码

  核心查询：递归 CTE 调用链

  -- get_call_graph 的核心 SQL（direction='callees', depth=2）
  WITH RECURSIVE call_chain(node_id, name, file_path, depth) AS (
      -- 起点
      SELECT n.id, n.name, f.path, 0
      FROM nodes n JOIN files f ON n.file_id = f.id
      WHERE n.name = ?1

      UNION ALL

      -- 递归展开
      SELECT n2.id, n2.name, f2.path, cc.depth + 1
      FROM call_chain cc
      JOIN edges e ON e.source_id = cc.node_id AND e.relation = 'calls'
      JOIN nodes n2 ON n2.id = e.target_id
      JOIN files f2 ON n2.file_id = f2.id
      WHERE cc.depth < ?2  -- depth 上限
  )
  SELECT * FROM call_chain ORDER BY depth;

  混合搜索：RRF 融合

  -- semantic_code_search 的 RRF 融合逻辑（伪SQL，实际在 Rust 中实现）
  -- Step 1: FTS5 搜索
  SELECT rowid, rank FROM nodes_fts WHERE nodes_fts MATCH ?1 ORDER BY rank LIMIT 20;

  -- Step 2: Vec0 向量搜索
  SELECT node_id, distance FROM node_vectors
  WHERE embedding MATCH ?embedding ORDER BY distance LIMIT 20;
